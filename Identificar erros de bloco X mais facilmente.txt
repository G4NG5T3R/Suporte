SELECT CRZ AS 'Contador da Redução Z',
CASE
WHEN MensagemWS LIKE '%ERRO 1001%'  THEN 'SCHEMA INVÁLIDO - GERE O ARQUIVO NOVAMENTE'
WHEN MensagemWS LIKE '%ERRO 2006%'  THEN 'ERRO 2006 - CREDENCIAMENTO ECF INFORMADO DIVERGE DO CADASTRADO NO S@T'
WHEN MensagemWS LIKE '%ERRO 3001%'  THEN 'ERRO 3001 - JÁ EXISTE CRZ ENVIADA COM ESTE NÚMERO PARA ESTE ECF'
WHEN MensagemWS LIKE '%ERRO 3010%'  THEN 'ERRO 3010 - GT INCONSISTENTE'
WHEN MensagemWS LIKE '%ERRO 3012%'  THEN 'ERRO 3012 - GT O VALOR DA VENDA BRUTA DEVE SER A SOMATÓRIA DE TODOS OS VALORES'
WHEN MensagemWS LIKE '%ERRO 3024%'  THEN 'ERRO 3024 - DATA DE REFERENCIA ANTERIOR A DATA DE INÍCIO DA OBRIGAÇÃO'
WHEN MensagemWS LIKE '%CERTIFICADO VÁLIDO%'  THEN 'CERTIFICADO DIGITAL NÃO LOCALIZADO NO CADASTRO DE LOJA'
WHEN MensagemWS LIKE '%AGUARDANDO RETORNO%'  THEN 'AGUARDANDO RETORNO DA SEFAZ PARA VALIDAÇÃO DO ARQUIVO'
END AS 'Retorno SEFAZ', DataMovimentoReducaoZ AS 'Data Movimento da Redução Z' FROM ControleEnvioArquivosPafEcf
WHERE Status<>5 ORDER BY CRZ ASC
